    function getMaxChoiceValue() {

      return Math.max(...scaleConfig.choices.map(c => c.value));

    }


    function collectAnswers() {

      const answers = {};

      let missing = [];


      scaleConfig.items.forEach((item, index) => {

        const checked = document.querySelector(`input[name="${item.id}"]:checked`);

        if (!checked) {

          missing.push(index + 1);

        } else {

          answers[item.id] = Number(checked.value);

        }

      });


      return { answers, missing };

    }


    function scoreAnswers(rawAnswers) {

      const max = getMaxChoiceValue();

      const scoredItems = [];

      const subscaleScores = {};

      let total = 0;


      scaleConfig.items.forEach(item => {

        const raw = rawAnswers[item.id];

        const scored = item.reverse ? (max + 1 - raw) : raw;


        scoredItems.push({

          id: item.id,

          text: item.text,

          raw,

          scored,

          subscale: item.subscale || "その他"

        });


        total += scored;

        const key = item.subscale || "その他";

        if (!subscaleScores[key]) {

          subscaleScores[key] = { total: 0, count: 0 };

        }

        subscaleScores[key].total += scored;

        subscaleScores[key].count += 1;

      });


      return {

        total,

        average: total / scaleConfig.items.length,

        scoredItems,

        subscaleScores

      };

    }

    }
